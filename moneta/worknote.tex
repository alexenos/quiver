%% Define Lists of ....
%% List of Processes
\newcommand{\listprocessname}{List of Processes}
\newlistof{process}{procs}{\listprocessname}

\newcommand{\process}[2]{%
  \refstepcounter{process}
  \subsubsection{Process: #1}\label{procs:#2}
  \addcontentsline{procs}{process}{\protect\numberline{\theprocess}#1}\par
}

%% List of Deliverables
\newcommand{\listdelivname}{List of Deliverables}
\newlistof{deliv}{delv}{\listdelivname}

\newcommand{\deliv}[2]{%
  \refstepcounter{deliv}
  \subsubsection{Delivery: #1}\label{deliv:#2}
  \addcontentsline{delv}{deliv}{\workdaydate \hspace{10 pt} #1}\par
}
